Select

All select controls in this documentation use the custom mngha select. It provides consistent styling, a selected check icon, controlled hover behavior, responsive layout, and theme support.

View Components

Overview

Consistent appearance

Every select uses the same trigger, listbox, option, selected check icon, and theme-aware colors.

Form compatible

A hidden input stores the selected value so regular form submission can include it.

Keyboard support

The supporting JavaScript handles opening, selection, arrows, Home, End, Enter, Space, Escape, and outside clicks.

Default Select

Use the custom select structure for every regular single-value selection field.

Required JavaScript

All custom selects require JavaScript. Add this once to the project; it initializes every element using data-custom-select.

JavaScript

Custom select behavior

function initCustomSelects() {
  const selects = document.querySelectorAll(
    "[data-custom-select]"
  );

  selects.forEach(function (select) {
    const trigger = select.querySelector(
      "[data-custom-select-trigger]"
    );

    const menu = select.querySelector(
      "[data-custom-select-menu]"
    );

    const value = select.querySelector(
      "[data-custom-select-value]"
    );

    const input = select.querySelector(
      "[data-custom-select-input]"
    );

    const options = Array.from(
      select.querySelectorAll(
        "[data-custom-select-option]"
      )
    );

    if (
      !trigger ||
      !menu ||
      !value ||
      !input ||
      options.length === 0
    ) {
      return;
    }

    function openSelect() {
      select.classList.add("is-open");
      trigger.setAttribute("aria-expanded", "true");

      const selected =
        options.find(function (option) {
          return (
            option.getAttribute("aria-selected") ===
            "true"
          );
        }) || options[0];

      selected.focus();
    }

    function closeSelect() {
      select.classList.remove("is-open");
      trigger.setAttribute("aria-expanded", "false");
    }

    function chooseOption(option) {
      options.forEach(function (item) {
        const isSelected = item === option;

        item.classList.toggle(
          "is-selected",
          isSelected
        );

        item.setAttribute(
          "aria-selected",
          String(isSelected)
        );

        item.tabIndex = isSelected ? 0 : -1;
      });

      value.textContent =
        option.textContent.trim();

      input.value =
        option.dataset.value || "";

      select.classList.remove(
        "mngha-custom-select--error"
      );

      trigger.removeAttribute("aria-invalid");

      closeSelect();
      trigger.focus();
    }

    trigger.addEventListener(
      "click",
      function () {
        if (
          select.classList.contains("is-open")
        ) {
          closeSelect();
        } else {
          openSelect();
        }
      }
    );

    trigger.addEventListener(
      "keydown",
      function (event) {
        if (
          event.key === "ArrowDown" ||
          event.key === "ArrowUp" ||
          event.key === "Enter" ||
          event.key === " "
        ) {
          event.preventDefault();
          openSelect();
        }

        if (event.key === "Escape") {
          closeSelect();
        }
      }
    );

    options.forEach(function (option, index) {
      option.addEventListener(
        "click",
        function () {
          chooseOption(option);
        }
      );

      option.addEventListener(
        "keydown",
        function (event) {
          if (
            event.key === "Enter" ||
            event.key === " "
          ) {
            event.preventDefault();
            chooseOption(option);
          }

          if (event.key === "ArrowDown") {
            event.preventDefault();

            options[
              (index + 1) % options.length
            ].focus();
          }

          if (event.key === "ArrowUp") {
            event.preventDefault();

            options[
              (
                index -
                1 +
                options.length
              ) % options.length
            ].focus();
          }

          if (event.key === "Home") {
            event.preventDefault();
            options[0].focus();
          }

          if (event.key === "End") {
            event.preventDefault();
            options[options.length - 1].focus();
          }

          if (event.key === "Escape") {
            closeSelect();
            trigger.focus();
          }
        }
      );
    });

    document.addEventListener(
      "click",
      function (event) {
        if (!select.contains(event.target)) {
          closeSelect();
        }
      }
    );
  });
}

if (document.readyState === "loading") {
  document.addEventListener(
    "DOMContentLoaded",
    initCustomSelects
  );
} else {
  initCustomSelects();
}

Class Reference

Class Element Purpose
.mngha-custom-select Container Provides positioning and open-state control.
.mngha-custom-select-trigger button Displays the current value and opens the option list.
.mngha-custom-select-arrow span Displays and rotates the dropdown arrow.
.mngha-custom-select-menu ul Displays the custom listbox menu.
.mngha-custom-select-option li Displays one selectable option.
.is-open Select container Shows the menu and rotates the arrow.
.is-selected Option Marks the selected option and shows its check icon.
.mngha-custom-select--error Select container Applies the error border and error focus state.
.mngha-field Container Groups the label, select, and supporting message.
.mngha-label Text Applies the field label typography.
.mngha-helper-text Text Displays supporting guidance.
.mngha-error-text Text Displays validation feedback.
.mngha-form-grid Container Arranges three fields and collapses on small screens.

Accessibility

Use a real trigger

The trigger must be a button with aria-haspopup="listbox" and an updated aria-expanded state.

Label the field

Connect the visible field label and selected value to the trigger using aria-labelledby.

Expose selection

Keep aria-selected synchronized with the visible selected check icon and the hidden input value.

Support the keyboard

Users must be able to open, move through, select, and close the options without a pointer.

Connect errors

Use aria-invalid and aria-describedby to associate validation feedback with the trigger.

Use unique IDs

Every label, selected-value element, and error message must have a unique id on the page.

Usage Guidance

Do

Keep labels concise, order options logically, use unique IDs, and initialize the component once after the page content loads.

Avoid

Do not mix native and custom select markup, omit keyboard handling, or forget to synchronize the hidden form value.

Initial value

When a value is preselected, update the visible text, option class, aria-selected, tabindex, and hidden input together.

Validation

Remove the error modifier after a valid selection and validate the hidden value again before processing the form.
